### 1. 大模型API的调用与流式响应处理

**面试题：如何在前端调用大模型API并处理流式响应？**

```javascript
// 使用fetch + ReadableStream实现流式处理
async function callModelAPI(prompt) {
  const response = await fetch('/api/generate', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ prompt })
  });

  const reader = response.body.getReader();
  const decoder = new TextDecoder();
  let accumulatedText = '';

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    accumulatedText += decoder.decode(value);
    console.log('实时接收:', accumulatedText);
  }
}
```

关键点：
- 使用fetch的流式响应模式
- 通过TextDecoder解码二进制数据
- 实时更新UI状态

### 2. AI功能的缓存与限流设计

**面试题：如何做AI功能的缓存与限流？**

```javascript
// 令牌桶限流算法实现（简化版）
class TokenBucket {
  constructor(options) {
    this.capacity = options.capacity;
    this.tokens = options.capacity;
    this.refillRate = options.refillRate;
    this.lastRefillTime = Date.now();
  }

  async request() {
    const now = Date.now();
    const elapsed = now - this.lastRefillTime;
    this.tokens = Math.min(this.capacity, this.tokens + (elapsed * this.refillRate) / 1000);
    this.lastRefillTime = now;

    if (this.tokens <= 0) {
      throw new Error('请求频率过高，请稍后再试');
    }
    this.tokens--;
    return true;
  }
}
```
